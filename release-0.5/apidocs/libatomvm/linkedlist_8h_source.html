76 static inline void linkedlist_remove(struct ListHead **list, struct ListHead *remove_item)
77 {
78  if (remove_item->next == remove_item) {
79  *list = NULL;
80  return;
81  }
82 
83  remove_item->prev->next = remove_item->next;
84  remove_item->next->prev = remove_item->prev;
85 
86  if (*list == remove_item) {
87  *list = remove_item->next;
88  }
89 }
