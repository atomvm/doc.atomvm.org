85 static inline term_ref ccontext_make_term_ref(struct CContext *ccontext, term t)
86 {
87  Context *ctx = ccontext->ctx;
88 
89  if (ctx->heap_ptr > ctx->e - 1) {
90  switch (memory_ensure_free(ctx, 1)) {
91  case MEMORY_GC_OK:
92  break;
93  case MEMORY_GC_ERROR_FAILED_ALLOCATION:
94  // TODO Improve error handling
95  fprintf(stderr, "Failed to allocate additional heap storage: [%s:%i]\n", __FILE__, __LINE__);
96  abort();
97  case MEMORY_GC_DENIED_ALLOCATION:
98  // TODO Improve error handling
99  fprintf(stderr, "Not permitted to allocate additional heap storage: [%s:%i]\n", __FILE__, __LINE__);
100  abort();
101  }
102  }
103 
104  ctx->e--;
105  ccontext->terms_count++;
106 
107  *ctx->e = t;
108  return ccontext->terms_count;
109 }
