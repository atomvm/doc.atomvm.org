66 static inline int atomvm_add_overflow(avm_int_t a, avm_int_t b, avm_int_t *res)
67 {
68  // a and b are shifted integers
69  avm_int_t sum = (a >> 4) + (b >> 4);
70  *res = sum << 4;
71  return ((sum > MAX_NOT_BOXED_INT) || (sum < MIN_NOT_BOXED_INT));
72 }

133 static inline int atomvm_mul_overflow_int64(avm_int64_t a, avm_int64_t b, avm_int64_t *res)
134 {
135  if ((a == 0) || (b == 0)) {
136  *res = 0;
137  return 0;
138 
139  } else {
140  avm_int64_t mul_res = a * b;
141  *res = mul_res;
142  return a != mul_res / b;
143  }
144 }

146 static inline int atomvm_mul_overflow(avm_int_t a, avm_int_t b, avm_int_t *res)
147 {
148  #if AVM_INT_MAX < INT64_MAX
149  avm_int64_t mul = (avm_int64_t) (a >> 2) * (avm_int64_t) (b >> 2);
150  *res = mul << 4;
151  return ((mul > MAX_NOT_BOXED_INT) || (mul < MIN_NOT_BOXED_INT));
152  #elif AVM_INT_MAX == INT64_MAX
153  int64_t mul;
154  int ovf = atomvm_mul_overflow_int64(a >> 2, b >> 2, &mul);
155  *res = mul << 4;
156  return ovf || ((mul > MAX_NOT_BOXED_INT) || (mul < MIN_NOT_BOXED_INT));
157  #else
158  #error "Unsupported AVM_INT_MAX size"
159  #endif
160 }
161 #endif
